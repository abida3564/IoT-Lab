const int ledPin= 12;
const int regPin = A0;
int sensorValue=0;



void setup()
{
  pinMode(regPin, INPUT);
  pinMode(ledPin, OUTPUT);
}

void loop()
{
  sensorValue= analogRead(regPin);
  
  digitalWrite(ledPin, HIGH);
  delay(sensorValue); 
  digitalWrite(ledPin, LOW);
  delay(sensorValue); 

}
